  /* ── Secondary · золото-охра (крона) ── */

  --gold-300:#E1CE89;  --gold-400:#D9C575;  --gold-500:#D3B961;
  --gold-600:#C0A544;  --gold-700:#A08636;  --gold-800:#7C6829; --gold-900:#5A4B1D;

  /* ── Роли ── */
  --color-primary:var(--green-500);         /* CTA, ссылки, навигация */
  --color-primary-hover:var(--green-700);
  --color-primary-active:var(--green-800);
  --color-secondary:var(--gold-500);        /* мегаменю, бейджи, акценты */

  /* ── Нейтрали ── */
  --bg:#FFFFFF; --bg-muted:#F7F7F5; --bg-sunken:#EFEEEA;
  --border:#E5E4E0; --border-strong:#D6D5CF;
  --text:#5B5B5B; --text-strong:#3A3A38; --text-muted:#8A8A85; --text-faint:#AFAEA8;
  --text-on-gold:#000000; --text-on-primary:#FFFFFF; --footer-bg:#0D4A2F;

  /* ── Семантика (цвет + иконка + текст) ── */
  --success:#18794D; --success-bg:#EAF3EE;   /* ✓ Одобрено */
